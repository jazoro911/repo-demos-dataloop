import numpy as np
from scipy.sparse import csr_matrix
from sklearn.decomposition import TruncatedSVD
from collections import defaultdict

class RecommendationEngine:
    """Motor de recomendaciones por filtrado colaborativo"""
    
    def __init__(self, n_factors=50):
        self.n_factors = n_factors
        self.model = None
        self.user_map = {}
        self.item_map = {}
        self.item_reverse_map = {}
    
    def fit(self, interactions_df):
        """Entrenar modelo con interacciones usuario-producto"""
        # Crear mappings
        users = interactions_df['customer_id'].unique()
        items = interactions_df['product_id'].unique()
        
        self.user_map = {u: i for i, u in enumerate(users)}
        self.item_map = {p: i for i, p in enumerate(items)}
        self.item_reverse_map = {i: p for p, i in self.item_map.items()}
        
        # Matriz sparse de interacciones
        rows = interactions_df['customer_id'].map(self.user_map)
        cols = interactions_df['product_id'].map(self.item_map)
        values = interactions_df['rating']
        
        self.matrix = csr_matrix(
            (values, (rows, cols)),
            shape=(len(users), len(items))
        )
        
        # Factorización SVD
        self.model = TruncatedSVD(n_components=self.n_factors)
        self.user_factors = self.model.fit_transform(self.matrix)
        self.item_factors = self.model.components_.T
        
        return self
    
    def recommend(self, customer_id, n=10, exclude_purchased=True):
        """Generar recomendaciones para un usuario"""
        if customer_id not in self.user_map:
            return self._cold_start_recommendations(n)
        
        user_idx = self.user_map[customer_id]
        scores = np.dot(self.user_factors[user_idx], self.item_factors.T)
        
        if exclude_purchased:
            purchased = self.matrix[user_idx].nonzero()[1]
            scores[purchased] = -np.inf
        
        top_indices = np.argsort(scores)[::-1][:n]
        
        recommendations = []
        for idx in top_indices:
            recommendations.append({
                'product_id': self.item_reverse_map[idx],
                'score': round(float(scores[idx]), 4)
            })
        
        return recommendations
    
    def _cold_start_recommendations(self, n):
        """Recomendaciones para usuarios nuevos (popularidad)"""
        popularity = np.array(self.matrix.sum(axis=0)).flatten()
        top_indices = np.argsort(popularity)[::-1][:n]
        return [{'product_id': self.item_reverse_map[i], 
                 'score': float(popularity[i])} for i in top_indices]

# Uso
import pandas as pd
df = pd.read_csv('user_product_ratings.csv')
engine = RecommendationEngine(n_factors=30)
engine.fit(df)
recs = engine.recommend('CUST-12345', n=5)
print("Recomendaciones:", recs)
